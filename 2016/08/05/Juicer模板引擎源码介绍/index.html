Juicer模板引擎源码介绍

  • Juicer
Javascript

Juicer是一款强大的、高校的前端页面模板引擎。它可以是页面元素与数据相分离。同时是一款开源代码的前端模板引擎，代码托管在Github上。

Juicer的详细使用及其API使用可以参考Juicer的官方中文网站，里面有Juicer详细的介绍和函数的使用，都是有中文翻译的，十分容易理解。

Juicer源代码解析

1、_escapehtml对象
_escapehtml对象是Juicer定义的内部对象，里面包括了<、>、&、”、’、/等特殊字符对应HTML中的表示符，及标识符替换方法、转换方法、数据检测方法。

var __escapehtml = {
    //特殊字符对应HTML中的表示符
    escapehash: {
        '<': '&lt;',
        '>': '&gt;',
        '&': '&amp;',
        '"': '&quot;',
        "'": '&#x27;',
        '/': '&#x2f;'
    },
    //标识符替换方法
    escapereplace: function(k) {
        return __escapehtml.escapehash[k];
    },
    //字符串替换其中特殊字符方法
    escaping: function(str) {
        return typeof(str) !== 'string' ? str : str.replace(/[&<>"']/igm, this.escapereplace);
    },
    //检查数据是否有值
    detection: function(data) {
        return typeof(data) === 'undefined' ? '' : data;
    }
};

2、_throw()异常打印
Juicer内部提供的异常打印方法，通过方法实现代码可看出，如果抛出异常直接控制台打印出错误日志。 

var __throw = function(error) {
    if(typeof(console) !== 'undefined') {
        if(console.warn) {
            console.warn(error);
            return;
        }

        if(console.log) {
            console.log(error);
            return;
        }
    }

    throw(error);
};

3、_creator()内部创建对象方法
Juicer提供了一个内部使用的创建对象的方法，并创建其属性。 

var __creator = function(o, proto) {
    o = o !== Object(o) ? {} : o;

    if(o.__proto__) {
        o.__proto__ = proto;
        return o;
    }

    var empty = function() {};
    var n = Object.create ? 
        Object.create(proto) : 
        new(empty.prototype = proto, empty);

    for(var i in o) {
        if(o.hasOwnProperty(i)) {
            n[i] = o[i];
        }
    }

    return n;
};

4、annotate(fn)获取方法的注解形式 

var annotate = function(fn) {
    var FN_ARGS = /^function\s*[^\(]*\(\s*([^\)]*)\)/m;
    var FN_ARG_SPLIT = /,/;
    var FN_ARG = /^\s*(_?)(\S+?)\1\s*$/;
    var FN_BODY = /^function[^{]+{([\s\S]*)}/m;
    var STRIP_COMMENTS = /((\/\/.*$)|(\/\*[\s\S]*?\*\/))/mg;
    var args = [],
        fnText,
        fnBody,
        argDecl;

    if (typeof fn === 'function') {
        if (fn.length) {
            fnText = fn.toString();
        }
    } else if(typeof fn === 'string') {
        fnText = fn;
    }

    fnText = fnText.trim();
    argDecl = fnText.match(FN_ARGS);
    fnBody = fnText.match(FN_BODY)[1].trim();

    for(var i = 0; i < argDecl[1].split(FN_ARG_SPLIT).length; i++) {
        var arg = argDecl[1].split(FN_ARG_SPLIT)[i];
        arg.replace(FN_ARG, function(all, underscore, name) {
            args.push(name);
        });
    }

    return [args, fnBody];
};

5、Juicer定义一些版本和初始化的东西
这里面是Juicer定义模板缓存、Juicer版本号、Juicer取值符号等，默认Juicer是开启模板缓存功能，这样是可以是模板编译一次以后就不用再次编译了，提高性能。并且定义了取值符${}或转义取值符&&{}等等。 

juicer.__cache = {};//缓存默认开启
juicer.version = '0.6.14';//版本号
juicer.settings = {};//相关设置信息
juicer.documentHTML = '';//

juicer.tags = {//标签定义
    operationOpen: '{@',
    operationClose: '}',
    interpolateOpen: '\\${',
    interpolateClose: '}',
    noneencodeOpen: '\\$\\${',
    noneencodeClose: '}',
    commentOpen: '\\{#',
    commentClose: '\\}'
};

juicer.options = {//初始化数据
    cache: true,
    strip: true,
    errorhandling: true,
    detection: true,
    _method: __creator({
        __escapehtml: __escapehtml,
        __throw: __throw,
        __juicer: juicer
    }, {})
};

6、juicer.set(conf, value)设置配置属性值
set()方法用以手动设置Juicer的配置属性值，或者设置操作标签值，如果要设置操作标签值，conf必须要tag::开头。 

juicer.set = function(conf, value) {
    var that = this;

    //全局替换\等特殊字符
    var escapePattern = function(v) {
        return v.replace(/[\$\(\)\[\]\+\^\{\}\?\*\|\.]/igm, function($) {
            return '\\' + $;
        });
    };

    //属性设置
    var set = function(conf, value) {
        var tag = conf.match(/^tag::(.*)$/i);

        if(tag) {
            that.tags[tag[1]] = escapePattern(value);
            that.tagInit();
            return;
        }

        that.options[conf] = value;
    };

    if(arguments.length === 2) {
        set(conf, value);
        return;
    }

    if(conf === Object(conf)) {
        for(var i in conf) {
            if(conf.hasOwnProperty(i)) {
                set(i, conf[i]);
            }
        }
    }
};

7、juicer.register(fname, fn)对象方法注册
第一个参数传入fname是模板对象或者是#id，fn是自定义函数。 

juicer.register = function(fname, fn) {
    //获取Juicer内部定义方法数组对象
    var _method = this.options._method;

    if(_method.hasOwnProperty(fname)) {
        return false;
    }

    //将自定义的方法添加进方法数组中
    return _method[fname] = fn;
};

8、juicer.unregister(fname)注销对象绑定方法
根据传入的方法名，将该方法从juicer定义的方法数组中移除。 

juicer.unregister = function(fname) {
    var _method = this.options._method;

    //直接从数组中删除该方法
    if(_method.hasOwnProperty(fname)) {
        return delete _method[fname];
    }
};

9、juicer.template(options)根据选项参数生成模板并返回模板对象

10、juicer.compile(tp1, options)根据模板对象及参数，编译模板但不渲染
第一个参数tp1是定义好的模板对象，options参数是数据集，此方法只将模板编译并放到内存中，并不会马山渲染。 

juicer.compile = function(tpl, options) {
    //获取数据集，并组装成苏剧对象。
    if(!options || options !== this.options) {
        options = __creator(options, this.options);
    }

    //设置缓存数据设置和取值方式
    var that = this;
    var cacheStore = {
        get: function(tpl) {
            if(options.cachestore) {
                return options.cachestore.get(tpl);
            }

            return that.__cache[tpl];
        },

        set: function(tpl, val) {
            if(options.cachestore) {
                return options.cachestore.set(tpl, val);
            }

            return that.__cache[tpl] = val;
        }
    };

    try {
        //将模板放到缓存中
        var engine = cacheStore.get(tpl) ? 
            cacheStore.get(tpl) : 
            new this.template(this.options).parse(tpl, options);

        if(!options || options.cache !== false) {
            cacheStore.set(tpl, engine);
        }

        return engine;

    } catch(e) {
        //异常处理
        __throw('Juicer Compile Exception: ' + e.message);

        return {
            render: function() {} // noop
        };
    }
};

11、juicer.to_html(tp1, data, options)将模板和对应数据生成HTML代码并返回 

juicer.to_html = function(tpl, data, options) {
    if(!options || options !== this.options) {
        options = __creator(options, this.options);
    }

    return this.compile(tpl, options).render(data, options._method);
};
